AssetManager Q - Demo.



Example Code

(function () {
    var load = true,
        assetManager = new AssetManager();

    function loadImages() {
        if (!load) {
            return;
        }
        load = false;

        var promise =
            assetManager
                .add([
                    {id: "first", src: "images/first.png"},
                    {id: "second", src: "images/second.gif"},
                    {id: "third", src: "images/third.jpg"}
                    ])
                .load();
        promise.then(function (assets) {
            for (var i = 0, l = assets.length; i < l; i++) {
                document.getElementById(assets[i].id).src = assets[i].src;
            }
        });
    }

    document.addEventListener("DOMContentLoaded", function() {
        document.getElementById("loadButton").addEventListener("click", loadImages);
    });
})();
